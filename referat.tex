\begin{frame}{Model Schöbel--Scholl: co optymalizujemy?}
  Dla każdej linii \(l \in \mathcal{L}\) szukamy jej częstotliwości \(f_l \in \mathbb{N}_{0}\), tak aby:
  \begin{itemize}
  \item każda relacja z \(R\) była obsłużona,
  \item łączny koszt nie przekraczał budżetu,
  \item podróże były możliwie wygodne dla pasażerów.
  \end{itemize}
  \begin{block}{Miara niewygody podróży}
    \[
      \Inc(P) = k_1\Time(P) + k_2\Trans(P)
    \]
    gdzie \(\Time(P)\) oznacza czas przejazdu, a \(\Trans(P)\) liczbę przesiadek.
  \end{block}
\end{frame}

\begin{frame}{Graf \emph{change \& go}}
  Aby modelować przesiadki, budujemy graf pomocniczy \(\mathcal{G} = (\mathcal{V}, \mathcal{E})\),
  \(\mathcal{V} = \mathcal{V}_{CG} \cup \mathcal{V}_{OD}\), \(\mathcal{E} = \mathcal{E}_{C} \cup \mathcal{E}_{G} \cup \mathcal{E}_{OD}\):
  \begin{itemize}
  \item \(\mathcal{V}_{CG}\): pary \emph{przystanek--linia},
    \formal{\(\mathcal{V}_{CG} = \{ (s, l) \in S \times \mathcal{L} : s \in V(l)\}\)} 
  \item \(\mathcal{E}_{G}\): przejazdy po jednej linii,
    \formal{\(\mathcal{E}_{l} = \{ ((s_1, l), (s_2, l)) \in \mathcal{V}_{CG} \times \mathcal{V}_{CG} : s_1s_2 \in E\}, \mathcal{E}_{G} = \bigcup_{l \in \mathcal{L}} \mathcal{E}_{l}\)}
  \item \(\mathcal{E}_{C}\): przesiadki w obrębie jednego przystanku,
    \formal{\(\mathcal{E}_{C} = \{ ((s, l_1), (s, l_2)) \in \mathcal{V}_{CG} \times \mathcal{V}_{CG} \}\)}
  \item \(\mathcal{V}_{OD}\): początki i końce podróży,
    \formal{\(\mathcal{V}_{OD} = \{ (s, \emptyset) : (s,t) \in R \lor (t,s) \in R\}\)}
  \item \(\mathcal{E}_{OD}\): wejście do sieci i wyjście z sieci.
    \formal{\(\mathcal{E}_{OD} = \bigcup_{\substack{(s,t) \in R, \, l \in \mathcal{L} \\ s,t \in V(l)}} 
    \Bigl\{ \bigl( (s, \varnothing), (s, l) \bigr), \, \bigl( (t, l), (t, \varnothing) \bigr) \Bigr\} \)}
  \end{itemize}
\end{frame}

\begin{frame}{Wagi krawędzi w grafie \emph{change \& go}}
  \[
    c_e =
    \begin{cases}
      0 & e \in \mathcal{E}_{OD} \\
      k_1 t_{uv} & e = ((u,l),(v,l)) \in \mathcal{E}_{G} \\
      k_2 & e \in \mathcal{E}_{C}
    \end{cases}
  \]
  \begin{itemize}
  \item \(t_{uv}\) -- czas przejazdu krawędzią \(uv\),
  \item \(k_1\) -- waga czasu,
  \item \(k_2\) -- koszt przesiadki.
  \end{itemize}
\end{frame}

\textcolor{olive} {
  Jeżeli interesuje nas tylko minimalizacja przesiadek, to możemy przyjąć, \(k_1 = 0, k_2 = 1\), jeżeli realistyczny czas przejazdu, to \(k_1 = 1\), a \(k_2\) to oszacowany czas przesiadki w tej samej jednostce, co czas pokonania krawędzi.
}

\begin{frame}{Ograniczenia w modelu Schöbel--Scholl}
  Model musi spełniać trzy grupy warunków:
  \begin{enumerate}
  \item \textbf{Obsługa popytu}: dla każdej relacji \((s,t) \in R\) istnieje poprawny przepływ.
  \item \textbf{Przepustowość}: liczba pasażerów nie przekracza pojemności i częstotliwości linii.
  \item \textbf{Budżet}: całkowity koszt eksploatacji mieści się w \(B\).
  \end{enumerate}
  Dodatkowo zmienne \(x_{st}^e\) i \(f_l\), reprezentujące przepływ pasażerów i częstotliwości linii, są całkowite.
\end{frame}

\begin{frame}{Model Schöbel--Scholl}
  Minimalizujemy sumaryczną niewygodę pasażerów na każdej trasie
  \small
  \[
    \min \sum_{(s,t) \in R} \sum_{e \in \mathcal{E}} c_e x_{st}^e
  \]
  przy ograniczeniach:
  \begin{itemize}
  \item równania przepływu dla każdej relacji,
  \item ograniczenia pojemności,
  \item ograniczenia infrastruktury,
  \item ograniczenie budżetowe,
  \item całkowitość zmiennych.
  \end{itemize}
\end{frame}

\begin{frame}{Przepustowość}
  \begin{enumerate}
  \item Pojazdy mają ograniczoną pojemność -- liczba pasażerów pokonujących każdy odcinek jest ograniczona przez sumaryczną pojemność pojazdów, które nim jadą.
    \formal{\(\displaystyle \forall l \in \mathcal{L} \, \forall e \in \mathcal{E}_{l} \: \sum_{(s,t) \in R} x_{st}^{e} \le N \cdot f_{l}\), \quad \(N\) -- pojemność jednego pojazdu}
  \item Liczba pojazdów przejeżdżających wzdłuż każdej krawędzi jest ograniczona.
    \formal{\(\displaystyle \forall k \in E \: \sum_{l \in \mathcal{L} : k \in \mathcal{E}_{l}}f_{l} \le f_{k}^{max} \), \quad \(f_{k}^{max}\) -- limit pojazdów dla krawędzi \(k\)
    }
  \end{enumerate}
\end{frame}
\begin{frame}{Budżet}

  Rozwiązanie musi spełnić ograniczenie \(\displaystyle \sum_{l \in \mathcal{L}} C_{l}f_{l} \le B\), gdzie:
  \begin{itemize}
  \item \(C_{l}\) -- koszt operacyjny 1 kursu na linii \(l\),
  \item \(B\) -- ustalony budżet.
  \end{itemize}

  Dodatkowo zmienne są całkowite:
  \[
    x_{st}^{e},f_{l} \in \mathbb{N} \quad \forall (s, t) \in R, e \in \mathcal{E}, l \in \mathcal{L} 
  \]
\end{frame}

\textcolor{olive} {   
Pełen Model
\begin{alignat*}{3}
  \min \quad & \sum_{(s, t) \in R} \sum_{e \in \mathcal{E}} c_{e}x_{st}^{e} && \quad \text{takie że:} && \\
             & x_{st}^{e} \le N \cdot f_{l} && \quad && \forall l \in \mathcal{L}, \forall e \in \mathcal{E}_{l} \\
             & \theta x_{st} = b_{st}       && \quad && \forall (s, t) \in R \\
             & \sum_{l \in \mathcal{L}} C_{l}f_{l} \le B && \quad && \\
             & \sum_{\substack{l \in \mathcal{L} \\ k \in \mathcal{E}_{l}}} f_{l} \le f_{k}^{\max} && \quad && \forall k \in E \\
             & x_{st}^{e}, f_{l} \in \mathbb{N} && \quad && \forall (s, t) \in R, e \in \mathcal{E}, l \in \mathcal{L}
\end{alignat*}
}

\section{Model Nachtigall--Jerosch}

\begin{frame}{Dlaczego drugi model?}
  Potrzebujemy modelu, który:
  \begin{itemize}
  \item lepiej oddaje zachowanie pasażerów,
  \item dopuszcza utratę części popytu,
  \item jest lżejszy obliczeniowo.
  \end{itemize}
\end{frame}

\begin{frame}{Model Nachtigall--Jerosch: główna idea}
  \begin{exampleblock}{Zmiana celu}
    Zamiast minimalizować niewygodę każdej możliwej podróży, mierzymy jej
    jakość i wymagamy, by była wystarczająco wysoka.
  \end{exampleblock}
  \begin{block}{Przewagi nad poprzednim modelem}
    \begin{itemize}
    \item Uwzględnia, że zbyt niewygodna podróż może zostać odrzucona przez pasażera.
    \item Ogranicza koszt pamięciowy i obliczeniowy reprezentacji sieci.
    \item Nie wymaga obsłużenia każdego pasażera.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Jakość drogi}
  \begin{block}{Definicja}
    \[
      \omega(p^{st}) = \rho t_{st}^* - \Time(p^{st}) - \beta\Trans(p^{st}).
    \]
  \end{block}
  \begin{itemize}
  \item \(t_{st}^*\) -- najkrótszy możliwy czas przejazdu,
  \item \(\rho\) -- współczynnik tolerancji wydłużenia,
  \item \(\beta\) -- kara za przesiadkę.
  \end{itemize}
  \vspace{0.2cm}
  Im większa wartość \(\omega\), tym atrakcyjniejsza podróż. Podróże o~ujemnej wartości \(\omega\) uznajemy za nieakceptowalne.
\end{frame}


\textcolor{olive} { 
Wagi krawędzi
\[
  \omega^{st}_{(ab), l} =
  \begin{cases}
    -\Time_{(ab), l} & a = s, b \ne t  \\
    -\Time_{(ab), l} - \beta & a \ne s, b \ne t \\
    \rho t_{st}^{*}- \Time_{(ab), l} - \beta & a \ne s, b = t \\
    \rho t_{st}^{*} - \Time_{(ab), l} & a = s, b = t
  \end{cases}
\]
}

\begin{frame}{Dwa poziomy optymalizacji w modelu Nachtigall--Jerosch}
  \begin{enumerate}
  \item \textbf{Traffic assignment}: rozdział pasażerów na akceptowalne ścieżki.
  \item \textbf{Line planning}: dobór częstotliwości linii przy ograniczeniach kosztu i infrastruktury.
  \end{enumerate}
\end{frame}

\begin{frame}{Traffic Assignment}
  \small
  Maksymalizujemy sumaryczną wygodę pasażerów.
  \[
    \max \: \sum_{(s, t) \in R} \sum_{l \in \mathcal{L}} \sum_{a, b \in V(l)} \omega^{st}_{(ab), l} x^{st}_{(ab), l} 
  \]
  przy ograniczeniach analogicznych, jak w modelu Sch\"obel-Scholl:
  \begin{itemize}
  \item Równania przepływu dla każdej relacji \((s, t) \in R\).
    \formal{\(\sum_{l \in \mathcal{L}} \sum_{a \in V(l)} x^{st}_{(at), l} \le w_{st}\)}
    \formal{\(\sum_{l \in \mathcal{L}} \sum_{a \in V(l)} x^{st}_{(av), l} = \sum_{l \in \mathcal{L}} \sum_{b \in V(l)} x^{st}_{(vb), l} \text{ dla } v \not \in \{s, t\} \)}
  \item<2> Przepływ pasażerów jest przybliżony liczbą rzeczywistą.
    \formal{\(x^{st}_{(ab), l} \in \mathbb{R}_{+}\)}
  \end{itemize}
\end{frame}

\begin{frame}{Line Planning}
  Ten etap wymaga uzupełnienia założeń i zmiennych poprzedniego
  \begin{itemize}
  \item Ograniczona pojemność pojazdów.
    \formal{\(\sum_{(s, t) \in R} \sum_{\substack{a, b \in V(l) \\ e \in l_{|ab}}} x^{st}_{(xb), l} \le Nf_{l} \quad \forall e \in E, l \in \mathcal{L}\)}
  \item Ograniczona przepustowość sieci.
    \formal{\(\sum_{\substack{l \in \mathcal{L} \\ e \in l}} f_{l} \le f^{max}(e) \quad \forall e \in E\)}
    \formal{\(\sum_{\substack{v \in \mathcal{V} \\ l \text{ serves } v}} f_{l} \le f^{max}(v) \quad \forall v \in V \)}
  \item Budżet.
    \formal{\(\sum_{l \in \mathcal{L}} C_{l}f_{l} \le B \)}
    \item Zmienne \(f_{l}\) przyjmuje wartości z \(\mathbb{N}\).
  \end{itemize}  
\end{frame}
